input = input.toLowerCase();
for(let i = 0; i<input.length;i++)
{

  {
    
    if(input[i] == vowels[j])
    {
      resultArray.push(input[i])
      if(input[i] == 'e' || input[i] == 'u')
      {resultArray.push(input[i])}
      

    }
  }

}
